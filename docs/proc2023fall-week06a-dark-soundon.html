# import required modulesdef procTags(istr):
# write your code here# --------------------------------------------------------------------# The following code is intended for your personal testing purposes.# Feel free to modify it to meet your requirements.# --------------------------------------------------------------------

def main():
    global data
    for i, td in enumerate(data, start=1):
