Exercise 2: ¬´Has the number been encountered before¬ª

numbers = [int(s) for s in input().split()]
occur_before = set()
for num in numbers:
    if num in occur_before:
        print('YES')
    else:
        print('NO')
        occur_before.add(num)
def check_encountered(numbers):
    encountered = []
    for num in numbers:
        if num in encountered:
            print("YES")
        else:
            print("NO")
            encountered.append(num)

nums = list(map(int, input().split()))
check_encountered(nums)

students = [{input() for j in range(int(input()))} for i in range(int(input()))]
known_by_everyone, known_by_someone = set.intersection(*students), set.union(*students)
print(len(known_by_everyone), *sorted(known_by_everyone), sep='\n')

def find_langs(n):
    al = set()
    aln = set()

    for i in range(n):
        nl = int(input())
        ll = []
        for _ in range(nl):
            ll.append(input())
        aln.update(ll)
        if i == 0:
            al.update(ll)
        else:
            al.intersection_update(ll)

    al = sorted(list(al))
    aln = sorted(list(aln))

    return len(al), al, len(aln), aln

ns = int(input())
na, a, nl, l = find_langs(ns)

print(na)
print(*a, sep='\n')
print(nl)
print(*l, sep='\n')

Exercise 4: ¬´The most frequent word¬ª

counter = {}
for i in range(int(input())):
    line = input().split()
    for word in line:
        counter[word] = counter.get(word, 0) + 1
        
max_count = max(counter.values())
most_frequent = [k for k, v in counter.items() if v == max_count]
print(min(most_frequent))
def count_words(line):
    freq = {}
    for _ in range(line):
        for word in input().split():
            try:
                freq[word] += 1
            except KeyError:
                freq[word] = 1
    return freq

def max_freq(freq):
    max_freq = max(freq.values())
    max_word = ''
    for word, f in freq.items():
        if f == max_freq:
            if word < max_word or max_word == '':
                max_word = word
    return max_word

num_lines = int(input())
words = count_words(num_lines)
print(max_freq(words))

from collections import defaultdict

latin_to_english = defaultdict(list)
for i in range(int(input())):
    english_word, latin_translations_chunk = input().split(' - ')
    latin_translations = latin_translations_chunk.split(', ')
    for latin_word in latin_translations:
        latin_to_english[latin_word].append(english_word)
    
print(len(latin_to_english))
for latin_word, english_translations in sorted(latin_to_english.items()):
    print(latin_word + ' - ' + ', '.join(english_translations))
def read_dictionary():
    n = int(input())
    dictionary = {}
    for i in range(n):
        entry = input().split(" - ")
        english_word = entry[0]
        latin_translations = entry[1].split(", ")
        for latin_word in latin_translations:
            add_to_dictionary(dictionary, latin_word, english_word)
    return dictionary

def add_to_dictionary(dictionary, latin_word, english_word):
    if latin_word in dictionary:
        dictionary[latin_word].append(english_word)
    else:
        dictionary[latin_word] = [english_word]

def print_dictionary(dictionary):
    print(len(dictionary))
    for latin_word in sorted(dictionary.keys()):
        english_words = ", ".join(sorted(dictionary[latin_word]))
        print(latin_word, "-", english_words)

dictionary = read_dictionary()
print_dictionary(dictionary)
