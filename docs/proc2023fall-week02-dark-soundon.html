Assignment Template

Template 1: No need to input data; the result is generated directly after processing.

#! /bin/python3

def main():
	print("Hello, Python world!")

if __name__=="__main__":
    main()
    exit(0)

#! /bin/python3

def drawTriangle(l):
    seq = range(1,l+1)
    if (l<0):
        seq = range(-l,0,-1)
    for i in seq:
        print("*"*i)
        
def main():
    drawTriangle(5)
    print()
    drawTriangle(-5)
    
if __name__=="__main__":
    main()
    exit(0)

Template 2 : Input integer/float/string data from the terminal and output the result after processing.

#! /bin/python3

def doSomething(a,b):
    print(a+b)

def main():
    a, b = [int(v) for v in input().split()]
    doSomething(a,b)
 
if __name__=="__main__":
    main()
    exit(0)
#! /bin/python3

def sum(nums):
    s = 0
    for n in nums:
        s+=n
    return s

def main():
    nums = [int(v) for v in input().split()]
    print(sum(nums))
 
if __name__=="__main__":
    main()
    exit(0)

Exercises and Assignments

Exercise 1: Please refer to Assignment Template 1 and print an equilateral triangle with 8 layers.

def drawRTriangle(l):
    for i in range(1,l+1):
        print(" "*(l-i)+"*"*(i*2-1))
        
def main():
    drawRTriangle(8)
    
if __name__=="__main__":
    main()
    exit(0)

Exercise 2: Please refer to Assignment Template 2, input a floating-point sequence from the terminal, and after comparison, print the maximum value in the sequence.

#! /bin/python3

def maxF(nums):
    m = nums[0]
    for f in nums[1:]:
        if f>m:
            m = f
    return m

def main():
    nums = [float(v) for v in input().split()]
    print(maxF(nums))
 
if __name__=="__main__":
    main()
    exit(0)
